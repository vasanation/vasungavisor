\beginverse*
Nikodemos, Nikodemos
var en höglärd man.
Han gick mitt i natten
för att söka katten.
Hem till någon, hem till någon
gick istället han.
\endverse

\beginverse*
När han bulta', när han bulta' 
steg den andre opp.
Knappt han öppnat dörren
åt främlingen förrän
frågor hagla', frågor hagla' 
mot hans huvudknopp.
\endverse

hos oss aldrig skådad.

är väl bara båg?"
\endverse

\beginverse*
Nikodemos, Nikodemos
ville veta mer
om den nya läran,
kom med en begäran:
"Gör ett under, gör ett under!
Tror gör den som ser."
\endverse
